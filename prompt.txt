create table public.tickets (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  title text not null,
  description text null,
  status public.ticket_status not null default 'Open'::ticket_status,
  priority public.ticket_priority not null default 'Medium'::ticket_priority,
  type public.ticket_type not null default 'Task'::ticket_type,
  due_date date null,
  is_blocked boolean null default false,
  assignee_id uuid null,
  sprint_id bigint null,
  organization_id uuid null,
  constraint tickets_pkey primary key (id),
  constraint tickets_assignee_id_fkey foreign KEY (assignee_id) references auth.users (id) on delete set null,
  constraint tickets_organization_id_fkey foreign KEY (organization_id) references organizations (id) on delete CASCADE,
  constraint tickets_sprint_id_fkey foreign KEY (sprint_id) references sprints (id) on delete set null
) TABLESPACE pg_default;
